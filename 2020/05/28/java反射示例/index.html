一、通过注解创建数据表

1、自定义创表注解

自定义设置属性的注解
package com.java.myAnnotation;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target(ElementType.FIELD)
@Retention(RetentionPolicy.RUNTIME)
public @interface MyField {
    //指定此Java属性在数据库的列名
    String columnName();
    //指定此Java属性在数据库中的字段长度
    int length() default 0;
    //设置此字段是否为空
    boolean isNull() default false;
    //设置此字段的数据类型
    String type();
    //设置此字段是否可以自增
    boolean isAutoIncrease();
    //设置此字段的默认值
    String Constraint() default "";
}

2、实体类

package com.java.bean;

import com.java.myAnnotation;
@MyTable("students")
public class Student {

    @MyField(columnName = "student_id", length = 0, type = " int", isAutoIncrease = true, Constraint = " primary key")
    private int id;

    @MyField(columnName="name", length=20, type=" varchar", isNull=true, isAutoIncrease = false , Constraint = " unique")
    private String name;

    @MyField(columnName="address", length=30, type=" varchar", isNull=true, isAutoIncrease = false)
    private String address;

    @MyField(columnName="phone", length=20, type=" varchar", isNull=true, isAutoIncrease = false , Constraint = " default('12345678910')")
    private String phone;

    public int getId() {
        return id;
    }

    public void setId(int id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public String getPhone() {
        return phone;
    }

    public void setPhone(String phone) {
        this.phone = phone;
    }

}

3、封装的创表工具类

package com.java.bean.utils;

import com.java.myAnnotation;
import java.lang.annotation.Annotation;
import java.lang.reflect.Field;

public class CreateSqlTable {
    public static String createSql(Class clazz) {
        String sql = null;

        if(clazz != null) {
            Annotation[] ans = clazz.getAnnotations();

            //遍历数据表注解
            for(Annotation an : ans) {
                if(an instanceof MyTable) {
                    //拼接创表语句
                    sql = "create table ";
                    MyTable myTable = (MyTable)an;
                    sql += myTable.value() + "(\n";
                    //获取实体类的所有属性
                    Field[] fields = clazz.getDeclaredFields();
                    //遍历实体类的所有属性
                    for(Field field : fields) {
                        //拿到每一个属性上的注解
                        Annotation[] annos = field.getAnnotations();
                        //遍历每一个属性上的注解
                        for(Annotation anno : annos) {
                            if(anno instanceof MyField) {
                                //开始拼接每一个字段的sql语句
                                MyField myField = (MyField)anno;
                                String autoIncrease = myField.isAutoIncrease() ? " identity" : "";
                                String length = myField.length() == 0 ? "" : "("+myField.length()+")";
                                String isNull = myField.isNull() ? "" : " not null";
                                sql += "\t" + myField.columnName() +""+ myField.type() +""+ length +""+ isNull +""+ autoIncrease +""+ myField.Constraint() +",\n";
                            }
                        }
                    }
                    sql = sql.substring(0, sql.length());
                    sql += ");";
                }
            }
            return sql;
        }
        return null;
    }
}

2、通过Java解析XML

package com.jm.utils;

import java.io.File;
import java.util.HashMap;
import javax.xml.parsers.DocumentBuilder;
import javax.xml.parsers.DocumentBuilderFactory;
import org.w3c.dom.Attr;
import org.w3c.dom.Document;
import org.w3c.dom.Element;
import org.w3c.dom.NamedNodeMap;
import org.w3c.dom.Node;
import org.w3c.dom.NodeList;
import org.w3c.dom.Text;

public class ParseXML {
    public static HashMap<String, String> sqlCon(String DbName) {
        HashMap<String, String> map = new HashMap<String, String>();
        //文档建立工厂
        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
        try {
            //文档建立器对象:负责解析和创建一个新的xml文档
            DocumentBuilder builder = factory.newDocumentBuilder();
            //解析文件或创建新的文件
            Document document = builder.parse(new File("sql_conn.xml"));

            //设置根节点
            NodeList node = document.getElementsByTagName("sql");

            //遍历此根节点
            for(int i=0; i<node.getLength(); i++) {
                Node num = node.item(i);
                if(!(num instanceof Text)) {
                    //获取此元素节点的属性 
                    NamedNodeMap attrs = num.getAttributes();
                    //遍历此元素节点的属性 
                    for(int k=0; k<attrs.getLength(); k++) {
                        Attr attr = (Attr)attrs.item(k) ;
                        if(DbName.equals(attr.getNodeValue())) {
                            //得到此节点下的所有的孩子节点
                            NodeList childNodes = node.item(i).getChildNodes();
                            //遍历所有的孩子节点
                            for(int j=0; j<childNodes.getLength(); j++) {
                                Node ch = childNodes.item(j) ;
                                if(!(ch instanceof Text)) {
                                    Element child = (Element)ch;
                                    //使用map集合存入
                                    map.put(child.getNodeName(), child.getLastChild().getNodeValue());
                                }
                            }
                        }
                    }
                }
            }
            return map;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return null;
    }
}

3、获取链接

package com.jm.utils;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;
import java.util.HashMap;

public class GetDBConnection {
    private static Connection con = null;

    public static Connection getCon() {
        //链接自己需要的数据库
        HashMap<String, String> map = ParseXML.sqlCon("sql2005");
        if(con == null) {
            try {
                //加载驱动
                Class.forName(map.get("driver"));
                //获取驱动链接
                String url = map.get("url");
                //设置用户名和密码
                con = DriverManager.getConnection(url, map.get("user"), map.get("pwd"));
            } catch (ClassNotFoundException e) {
                // TODO Auto-generated catch block
                e.printStackTrace();
            } catch (SQLException e) {
                // TODO Auto-generated catch block
                e.printStackTrace();
            }
        }
        return con;
    }
}

一、循环遍历集合

1、遍历排序list集合

package com.blog.lambda;

import com.blog.bean.Person;

import java.util.ArrayList;
import java.util.List;

public class LambdaDemo03 {
    public static void main(String[] args) {
        List<Person> list = new ArrayList<>();
        list.add(new Person("zs", 12));
        list.add(new Person("ls", 14));
        list.add(new Person("ss", 17));
        list.add(new Person("ks", 11));
        list.add(new Person("as", 19));
        list.add(new Person("hs", 15));
        list.add(new Person("ds", 10));

        //利用list的sort对person对象的age进行排序
        list.sort((o1,o2) -> o2.getAge()-o1.getAge());
        //遍历集合
        list.forEach(person -> System.out.println(person.getName() + person.getAge()));


        List<Integer> list1 = new ArrayList<>();
        list1.add(1);
        list1.add(5);
        list1.add(2);
        list1.add(7);
        list1.add(3);
        list1.add(9);
        list1.add(2);
        list1.sort(Integer::compareTo);

        list1.forEach(integer -> {System.out.println(integer);});
    }
}

2、遍历排序set集合

package com.blog.lambda;

import com.blog.bean.Person;
import java.util.Set;
import java.util.TreeSet;

public class LambdaDemo03 {
    public static void main(String[] args) {
        Set<Person> set = new TreeSet<>(((o1, o2) -> {
            if (o1.getAge() >= o2.getAge()) {
                return 1;
            }else {
                return -1;
            }
        }));

        set.add(new Person("zs", 12));
        set.add(new Person("ls", 14));
        set.add(new Person("ss", 17));
        set.add(new Person("ks", 11));
        set.add(new Person("as", 19));
        set.add(new Person("hs", 15));
        set.add(new Person("ds", 10));
        set.forEach(person -> System.out.println(person.getName() + person.getAge()));
    }
}

四、闭包函数

1、闭包示例一：

package com.blog.lambda;

import java.util.function.Supplier;

public class LambdaDemo03 {
    public static void main(String[] args) {
        /**
         * 由于使用闭包函数，在getNum方法执行完了以后，局部变量并没有被销毁
         * 而且后面Supplier<Integer>的get方法还对num进行了引用
         * 由此可知使用闭包函数会提升局部变量的生命周期
         */
        int num = getNum().get();
        System.out.println(num);
    }

    private static Supplier<Integer> getNum() {
        //局部变量 ， 方法执行完销毁
        int num = 10;

        //因为返回值是Supplier<Integer>，而Supplier<Integer>的返回值是Integer，所以直接返回num
        return () -> num;
    }
}

2、闭包示例二：

package com.blog.lambda;

import java.util.function.Consumer;
import java.util.function.Supplier;

public class LambdaDemo03 {
    public static void main(String[] args) {
        int num = 1;
        //正常输出
        Consumer<Integer> consumer01 = T-> System.out.println(T);
        consumer01.accept(23);

        /**
         * 直接输出局部变量的值
         * 如果使用的是这种方式,应该将局部变量设置为常量
         * 因为使用lambda表达式在闭包中引用的变量都是常量
         * 如果我们不加final修饰使其变为常量的话，编译器会默认的加上final修饰
         * 此时我们是无法对这个局部变量进行修改的
         */
        Consumer<Integer> consumer02 = T-> System.out.println(num);

        //因为输出的是一个局部变量，所以无论我们传的值是多少，输出的永远是局部变量
        consumer02.accept(23);
    }
}
